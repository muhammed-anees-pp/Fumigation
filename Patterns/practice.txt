Program:
#include <stdio.h>

int main() {
    int n = 6;

    for (int i = 1; i <= n; i++) {
        for (int j = 1; j <= i; j++) {
            if (j % 2 == 1) {
                printf("%d ", i * j);
            } else {
                printf("* ");
            }
        }
        printf("\n");
    }

    return 0;
}
