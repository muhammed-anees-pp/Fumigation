Q:

    *
   * *
  * * *
 * * * *
* * * * *

Program:

int i,j,row=5;
for(i=1;i<=row;i++){
for(j=1;j<=row;j++){
if((i+j)<=row) printf(" ");
else printf("* ");
}
printf("\n");
}
